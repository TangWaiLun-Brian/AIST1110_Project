class Obstacle(Rectangle):
    def __init__(self):
        width = random.randint(9, 60)
        height = random.randint(9, 60)
        centerx = random.randint(-1, SCREEN_WIDTH - width)
        centery = random.randint(-1, 390 - height)
        super(Obstacle, self).__init__((centerx, centery), width, height)


def check_collison(obstacle, ball):
    flag = False
    if obstacle.rect.left <= ball.rect.right and obstacle.rect.right >= obstacle.rect.right and obstacle.rect.top <= ball.rect.centery <= obstacle.rect.bottom:
        ball.speed[-1] *= -1
        flag = True
    if obstacle.rect.right >= ball.rect.left and obstacle.rect.left <= obstacle.rect.right and obstacle.rect.top <= ball.rect.centery <= obstacle.rect.bottom:
        ball.speed[-1] *= -1
        flag = True
    if obstacle.rect.top <= ball.rect.bottom and obstacle.rect.bottom >= obstacle.rect.top and obstacle.rect.left <= ball.rect.centerx <= obstacle.rect.right:
        ball.speed[0] *= -1
        flag = True
    if obstacle.rect.bottom >= ball.rect.top and obstacle.rect.top <= obstacle.rect.bottom and obstacle.rect.left <= ball.rect.centerx <= obstacle.rect.right:
        ball.speed[0] *= -1
        flag = True
    return flag
